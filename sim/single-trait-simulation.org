### Step 1. generate causal snplists
repNum = 1
h2snp = 0.5
sams <- c(10) * 1000
sigmaSq = h2snp / sams
samsLeft <- bim
tmp <- samsLeft[sample(.N,sams)]
tmp$beta = rnorm(sams,0,sqrt(sigmaSq) )

causalSnplistFile = paste0(outPath,rawBimFileSuffix,"-causal.snplist")
fwrite(data.frame(tmp[,c("SNP","beta")]),file=causalSnplistFile,col.names = FALSE, sep = "\t")
### Step 2. do simulation
gctaMlm <- paste0(gcta64," --bfile ",rawDataPath,rawBimFileSuffix,
                  " --simu-qt ",
                  " --simu-causal-loci ", causalSnplistFile,
                  " --simu-hsq ",h2snp,
                  " --simu-rep ", repNum,
                  " --thread-num 100",
                  " --out ", outPath, rawBimFileSuffix, ".sim"
                  )
system(gctaMlm)

## Step 3. Calculate allele frequency
plinkFreq <- paste0(plink19," --bfile ",rawDataPath,rawBimFileSuffix,
                    " --freq ",
                    " --out ", rawDataPath,rawBimFileSuffix
                    )

## Step 4. extract summary data
plinkGWAS <- paste0(plink19," --bfile ",rawDataPath,rawBimFileSuffix,
                    " --pheno ", outPath, rawBimFileSuffix, ".sim.phen",
                    " --linear ",
                    " --ci 0.95 ",
                    " --out ", outPath, rawBimFileSuffix, ".assoc"
                    )

## run gwas function

## flip allele frequency
assocCom[bA1A2,A2:= A1_ref]
assocCom[bA1A2, FRQ_ref:=1-FRQ_ref]
