library(data.table)
mainPath = "/home/uqsliu28/90days/projects/sbayesRC/ldAndMa/ldblockComm/"
rawDatPath = "/home/uqsliu28//projects/annotation//dataRaw/ukbGeno/"
ldbinfoFileSuffix = "chr1-22-ukbEURu_imp_v3_HM3_maf01.eigen.ldblock.info"
snpinfoFileSuffix = "chr1-22-ukbEURu_imp_v3_HM3_maf01.eigen.ldblock.snp.info"
ldbinfo = fread( paste0(rawDatPath,ldbinfoFileSuffix))
snpinfo = fread( paste0(rawDatPath,snpinfoFileSuffix))
lbs = unique(ldbinfo$LDBLOCK)
snpDup = ldbinfo$snpInLdBlock[duplicated(ldbinfo$snpInLdBlock)]
message("there are total ", length(snpDup)," snp duplicated.")

snpInfoDup = snpinfo[snpinfo$ID %in% snpDup,]
lbsSnpDup = merge(snpInfoDup, ldbinfo, by.x = "ID", by.y = "snpInLdBlock")
## decide which SNPs are located a given LD block based on [,)
removedIdx <- c()
for(snp in snpDup){
  dup = lbsSnpDup[which(lbsSnpDup$ID == snp),]
  removedLbs = max(dup$LDBLOCK)
  idx = which(ldbinfo$LDBLOCK == removedLbs & ldbinfo$snpInLdBlock == snp)
  removedIdx <- c(removedIdx, idx)
}

ldbinfoUniq = ldbinfo[- removedIdx,]


for(i in lbs){
  snps = ldbinfoUniq[which(ldbinfoUniq$LDBLOCK == i),]
  snpNumInLbs = unique(snps$NumSnpInLdBlock)
  snplist = snps$snpInLdBlock
  outFile = paste0(mainPath,"/snplist/","chr1-22-ukbEURu_imp_v3_HM3_maf01-ldm-block-",i,".snplist")
  fwrite(data.frame(snplist),file = outFile,col.names = FALSE)
}

outFile = paste0(mainPath,"/snplist/","chr1-22-ukbEURu_imp_v3_HM3_maf01")

###############################################
#### Step 2. generate residual  ####
###############################################
h21 = 0.7
h22 = 0.3
## Step 2.1  dealing with beta
trait1 = fread(paste0(causalsnpsFile, "-trait-1.gcta.phen"))
trait2 = fread(paste0(causalsnpsFile, "-trait-2.gcta.phen"))
names(trait1) <- c("FID","IID","beta")
names(trait2) <- c("FID","IID","beta")
trait1[, betaS:= beta/sd(beta) * sqrt(h21)]
trait2[, betaS:= beta/sd(beta) * sqrt(h22)]

## Step 2.2 simulate bivariate residual effect based on bivariate normal distribution
mu = c(0,0)
covMat = matrix(c(0.3, 0.235, 0.235, 0.8), ncol = 2)
indis = 300 * 1000

bivRes <- mvrnorm(n = indis, mu = mu, Sigma = covMat)

trait1$res = bivRes[,1]
trait2$res = bivRes[,2]

trait1[,resS:= res/sd(res) * sqrt(1- h21)]
trait2[,resS:= res/sd(res) * sqrt(1- h22)]

## Step 2.3 phenotype
trait1[,y:= betaS + resS]
trait2[,y:=betaS + resS]

trait1[,yS:=(y-mean(y))/sd(y)]
trait2[,yS:=(y-mean(y))/sd(y)]

traitphenFile1 = paste0(causalsnpsFile, "-trait-1.gcta.phen.plink")
